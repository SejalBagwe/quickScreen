package com.example.demo.service;
import java.time.Instant;

import org.springframework.stereotype.Service;

@Service
public class DemoService{

public void calculateHash() {
		String s=String.format("%.3f",(double)Instant.now().toEpochMilli()/1000);
		System.out.println("That is, your generated hash output should look something like this:");
		System.out.println(s+"-"+calculateSum(s.substring(0, s.indexOf(".")))
				+"-"+ calculateSum(s.substring(s.indexOf(".")+1, s.length())));
	}
	
	public Integer calculateSum(String s) {
		Integer sum=0;
		 for(int i=0;i<s.length();i++){
			 sum=sum+s.charAt(i) - '0';
		}
		 return sum;
		 
	}
}
